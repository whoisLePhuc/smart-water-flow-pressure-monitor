cmake_minimum_required(VERSION 3.22)

# =============================================================================
# Project definition
# =============================================================================
set(CMAKE_PROJECT_NAME firmware)
project(${CMAKE_PROJECT_NAME} C ASM)

set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS TRUE)

if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE "Debug")
endif()

message("Build type: " ${CMAKE_BUILD_TYPE})

# =============================================================================
# Firmware executable
# =============================================================================
# Phải tạo executable TRƯỚC khi add_subdirectory(stm32cubemx) vì file đó
# dùng target_sources() để thêm main.c, startup, HAL sources vào target này.
add_executable(${CMAKE_PROJECT_NAME})

# =============================================================================
# Modules
# =============================================================================

# Vendor STM32 HAL + Core system sources (generated by STM32CubeMX)
# Handles: CMSIS includes, HAL sources, HAL config, system files, startup
add_subdirectory(cmake/stm32cubemx)

# Hardware driver modules
add_subdirectory(Drivers)

# Test modules (optional, enable via -DFIRMWARE_BUILD_TESTS=ON)
add_subdirectory(Core/Tests)

# Link all modules into firmware
# Dùng plain signature để tương thích với cmake/stm32cubemx/CMakeLists.txt
target_link_libraries(${CMAKE_PROJECT_NAME}
    drivers
    stm32cubemx
)

# Test targets (optional — bật bằng -DFIRMWARE_BUILD_TESTS=ON)
if(FIRMWARE_BUILD_TESTS)
    # Truyền define xuống C code để #ifdef FIRMWARE_BUILD_TESTS hoạt động
    target_compile_definitions(${CMAKE_PROJECT_NAME}
        PRIVATE
            FIRMWARE_BUILD_TESTS=1
    )

    if(FIRMWARE_BUILD_TESTS_MAX35103)
        target_compile_definitions(${CMAKE_PROJECT_NAME}
            PRIVATE
                FIRMWARE_BUILD_TESTS_MAX35103
        )
    endif()

    if(FIRMWARE_BUILD_TESTS_FM24CL04B)
        target_compile_definitions(${CMAKE_PROJECT_NAME}
            PRIVATE
                FIRMWARE_BUILD_TESTS_FM24CL04B
        )
    endif()

    if(FIRMWARE_BUILD_TESTS_ZSSC3241)
        target_compile_definitions(${CMAKE_PROJECT_NAME}
            PRIVATE
                FIRMWARE_BUILD_TESTS_ZSSC3241
        )
    endif()

    if(FIRMWARE_BUILD_TESTS_FM24CL04B)
        target_link_libraries(${CMAKE_PROJECT_NAME} test_fram)
    endif()
    if(FIRMWARE_BUILD_TESTS_MAX35103)
        if(TARGET driver_max35103_stm32_hal)
            target_link_libraries(${CMAKE_PROJECT_NAME} test_max35103 driver_max35103_stm32_hal)
        else()
            target_link_libraries(${CMAKE_PROJECT_NAME} test_max35103)
        endif()
    endif()
    if(FIRMWARE_BUILD_TESTS_ZSSC3241)
        if(TARGET driver_zssc3241_stm32_hal)
            target_link_libraries(${CMAKE_PROJECT_NAME} test_zssc3241 driver_zssc3241_stm32_hal)
        else()
            target_link_libraries(${CMAKE_PROJECT_NAME} test_zssc3241)
        endif()
    endif()
endif()

# MAX35103 AutoCal service (optional — bật bằng -DFIRMWARE_BUILD_MAX35103_AUTOCAL=ON)
option(FIRMWARE_BUILD_MAX35103_AUTOCAL "Build MAX35103 AutoCal service into firmware" ON)

if(FIRMWARE_BUILD_MAX35103_AUTOCAL)
    target_compile_definitions(${CMAKE_PROJECT_NAME}
        PRIVATE
            FIRMWARE_BUILD_MAX35103_AUTOCAL
    )

    add_subdirectory(Core/Src/calibrate)
    target_link_libraries(${CMAKE_PROJECT_NAME} calibrate)
endif()

# Remove wrong libob.a library dependency
list(REMOVE_ITEM CMAKE_C_IMPLICIT_LINK_LIBRARIES ob)
